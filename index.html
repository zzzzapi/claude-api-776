# ChatGPT API调用示例 (Python)
import requests

# 设置API密钥和中转接口地址
API_KEY = "your_4z_api_key"  # 在控制台获取您的API密钥
API_URL = "https://api.zzzzapi.com/v1/chat/completions"  # 中转接口地址

# 设置请求头

    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

# 设置请求参数
data = {
    "model": "gpt-4o",  # 可选择多种模型：gpt-4o, gpt-4, gpt-3.5-turbo等
    "messages": [
        {"role": "system", "content": "你是一个专业助手"},
        {"role": "user", "content": "请解释API中转站的作用"}
    ],
    "temperature": 0.7,  # 控制创造性，0-1之间
    "max_tokens": 1000   # 控制回复长度
}

# 发送请求并获取响应
response = requests.post(API_URL, headers=headers, json=data)
print(response.json()["choices"][0]["message"]["content"])
